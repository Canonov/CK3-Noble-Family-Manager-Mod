nfm_housetype_steward = 1
nfm_housetype_martial = 2
nfm_housetype_learning = 3
nfm_housetype_intrigue = 4
nfm_housetype_diplomatic = 5
nfm_housetype_foreign = 6


nfm_value_gethousetype = {
    value = 0

    if = {
        limit = { scope:nfm_house_steward ?= yes }
        value = nfm_housetype_steward
    }
    else_if = {
        limit = { scope:nfm_house_martial ?= yes }
        value = nfm_housetype_martial
    }
    else_if = {
        limit = { scope:nfm_house_learning ?= yes }
        value = nfm_housetype_learning
    }
    else_if = {
        limit = { scope:nfm_house_intrigue ?= yes }
        value = nfm_housetype_intrigue
    }
    else_if = {
        limit = { scope:nfm_house_diplomatic ?= yes }
        value = nfm_housetype_diplomatic
    }
    else_if = {
        limit = { scope:nfm_house_foreign ?= yes }
        value = nfm_housetype_foreign
    }
}

    value = 0

    switch = {
        trigger = scope:family_size

        nfm_familysize_small = { add = { 2 4 } }

        nfm_familysize_large = { add = { 2 3 } }
    }
}
