
## nfm_effect_create_house_steward
# Creates a complete stewardship-focused noble family with parents, house head,
# heir, and siblings. Family size varies based on the family_size scope value.
#
# @scope: character - Must be called from the liege/employer who will hire the family 
# @input scope:family_size [value] - Controls family composition (small/medium/large)
# @output scope:new_house_head [character] - The generated family head
# @example:
#   scope:family_size = nfm_familysize_medium
#   root = { nfm_script_create_family_stewards = yes }
nfm_effect_create_house_steward = {

    # Generate Parents if it's not a small family
    if = {
        limit = {
            scope:family_size > nfm_familysize_small
        }

        # Mother and Father
        create_character = {
            template = pool_repopulate_stewardship
            gender = male
            age = { 60 95 }
            culture = root.culture
            faith = root.faith
            employer = root

            dynasty = generate
            save_scope_as = head_father
        }
        create_character = {
            template = pool_repopulate_stewardship
            gender = female
            age = { 60 80 }
            culture = root.culture
            faith = root.faith
            employer = root

            dynasty = none

            save_scope_as = head_mother
        }

        scope:head_father = { marry = scope:head_mother }

        # House Head
        create_character = {
            template = nfm_character_familyhead_stewardship
            father = scope:head_father
            mother = scope:head_father.primary_spouse
            employer = root

            dynasty = inherit

            save_scope_as = new_house_head
        }
    }
    else = {
        create_character = {
            template = nfm_character_familyhead_stewardship
            employer = root

            dynasty = generate

            save_scope_as = new_house_head
        }
    }

    # Generate an Adult Heir
    create_character = {
        template = nfm_character_familyhead_stewardship
        age = { 16 18 }
        employer = root

        father = scope:new_house_head

        dynasty = inherit

        save_scope_as = new_heir
    }

    while = {
        count = nfm_familyhead_extrachildcount

        create_character = {
            template = nfm_character_child_stewardship
            employer = root

            age = { 0 nfm_value_oneyoungerthanheir }

            father = scope:new_house_head

            dynasty = inherit
        }
    }


    # Generate Siblings
    if = {
        limit = {
            scope:family_size > nfm_familysize_small
        }

        while = {
            count = nfm_familysize_siblingsforcount

            create_character = {
                template = nfm_character_sibling_stewardship
                age = { 25 nfm_value_new_house_head_age }
                employer = root

                father = scope:head_father
                mother = scope:head_mother

                dynasty = inherit
            }
        }
    }

}
