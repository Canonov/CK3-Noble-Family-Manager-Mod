
# Elevate a Noble Family
nfm_decision_elevate_nobles = {
    title = {
        first_valid = {
            # Elevate a Minor Clan
            triggered_desc = {
                trigger = {
                    OR = {
                        culture = {
                            OR = {
                                # Japanese
                                has_cultural_pillar = heritage_japonic
                                # Other Clan Preferring Cultures
                                has_cultural_pillar = heritage_arabic
                                has_cultural_pillar = heritage_iranian
                                has_cultural_pillar = heritage_turkic
                            }
                        }
                        has_religion = religion:islam_religion
                    }
                }
                desc = nfm_decision_elevate_nobles_clan
            }
            # Default: Elevate a Noble Family
            desc = nfm_decision_elevate_nobles
        }
    }

    desc = nfm_decision_elevate_nobles_desc
    selection_tooltip = nfm_decision_elevate_nobles_tooltip

    picture = {
        trigger = { culture = { has_graphical_east_asia_culture_group_trigger = yes } }
        reference = "gfx/interface/illustrations/decisions/tgp_scholars.dds"
    }
    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_recruitment.dds"
    }

    decision_group_type = courtier

    is_shown = {
        # for now, maybe
        is_ai = no

        # Must be at least a King
        is_landed = yes
        highest_held_title_tier >= tier_kingdom

        # Check Government
        government_allows = administrative
    }

    is_valid = {
    }

    widget = {
        gui = "decision_view_widget_option_list_generic"
        controller = decision_option_list_controller
        decision_to_second_step_button = "nfm_decision_elevate_nobles_confirm"
        show_from_start = yes # No would make it open the list after

        # House of Administrators
        item = {
            value = nfm_house_steward

            localization = nfm_house_steward_name
            current_description = nfm_house_steward_desc

            icon = "gfx/interface/icons/icon_skill_stewardship.dds"
            is_default = yes

            ai_chance = {
                value = 0
            }
        }

        # House of Warriors
        item = {
            value = nfm_house_martial

            localization = {
                first_valid = {
                    # Japanese - Samurai
                    triggered_desc = {
                        trigger = { culture = { has_cultural_pillar = heritage_japonic } }
                        desc = nfm_house_martial_name_samurai
                    }
                    desc = nfm_house_martial_name
                }
            }
            current_description = nfm_house_martial_desc

            icon = "gfx/interface/icons/icon_knight_acclaimed.dds"

            ai_chance = {
                value = 0
            }
        }

        # House of Scholars
        item = {
            value = nfm_house_learning

            localization = {
                first_valid = {
                    # Chinese - Confucian Scholars
                    triggered_desc = {
                        trigger = {
                            OR = {
                                culture = { has_cultural_tradition = tradition_scholar_officials }
                                faith.religion = religion:confucianism_religion
                            }
                        }
                        desc = nfm_house_learning_name_confucian
                    }
                    desc = nfm_house_learning_name
                }
            }
            current_description = {
                desc = nfm_house_learning_desc
            }

            icon = "gfx/interface/icons/icon_skill_learning.dds"

            ai_chance = {
                value = 0
            }
        }

        # House of Spies
        item = {
            value = nfm_house_intrigue

            localization = nfm_house_intrigue_name
            current_description = nfm_house_intrigue_desc

            icon = "gfx/interface/icons/icon_scheme.dds"

            ai_chance = {
                value = 0
            }
        }

        # House of Foreigners
        item = {
            value = nfm_house_foreign

            localization = nfm_house_foreign_name
            current_description = nfm_house_foreign_desc
            is_valid = {
                any_realm_county = { culture != root.culture }
            }

            icon = "gfx/interface/icons/icon_culture.dds"

            ai_chance = {
                value = 0
            }
        }

    }

    effect = {
        ## Choose a House Size
        hidden_effect = {
            random_list = {
                20 = {
                    save_scope_value_as  = {
                        name = family_size
                        value = nfm_familysize_small
                    }
                }
                60 = {
                    save_scope_value_as  = {
                        name = family_size
                        value = nfm_familysize_medium
                    }
                }
                20 = {
                    save_scope_value_as  = {
                        name = family_size
                        value = nfm_familysize_large
                    }
                }
            }
        }

        custom_tooltip = nfm_decision_elevate_nobles_sizes_tt

        ## Save the House Type Once
        save_scope_value_as = {
            name = house_type
            value = nfm_value_gethousetype
        }

        ## Create the Family
        hidden_effect = {
            switch = {
                trigger = scope:house_type
                nfm_housetype_steward = {
                    nfm_effect_create_house_steward = { EMPLOYER = root }
                }
                nfm_housetype_martial = {
                    nfm_effect_create_house_martial = { EMPLOYER = root }
                }
                nfm_housetype_learning = {
                    nfm_effect_create_house_learning = { EMPLOYER = root }
                }
                nfm_housetype_intrigue = {
                    nfm_effect_create_house_intrigue = { EMPLOYER = root }
                }
                nfm_housetype_foreign = {
                    nfm_effect_create_house_foreign = { EMPLOYER = root }
                }
            }

            scope:new_house_head.house = {
                set_variable = {
                    name = nfm_housetype
                    value = scope:house_type
                }
                set_variable = {
                    name = nfm_familysize
                    value = scope:family_size
                }
            }

            # Grant portion of influence cost to new family head
            scope:new_house_head = {
                change_influence = 200
            }
        }

        # Will be created tooltip
        custom_tooltip = nfm_decision_elevate_nobles_tt

        # Tooltip to explain their culture and faith
        if = {
            limit = { scope:house_type = nfm_housetype_foreign }
            custom_tooltip = nfm_house_foreign_diff_culture_tooltip
        }
        # Tooltip to explain merit granting
        if = {
            limit = { government_allows = merit }
            custom_tooltip = nfm_decision_elevate_nobles_getmerit_tt
        }

        # Trigger the event of their arrival
        trigger_event = {
            id = nfm_events_elevate_nobles.1
        }
    }

    cost = {
        prestige = {
            value = 200
        }

        gold = {
            value = nfm_value_elevate_cost_gold
        }

        treasury = {
            value = nfm_value_elevate_cost_treasury
        }

        influence = {
            value = 400
        }
    }

    # Prevent AI from taking this decision
    ai_potential = {
        always = no
    }

    ai_check_interval = 0
}
