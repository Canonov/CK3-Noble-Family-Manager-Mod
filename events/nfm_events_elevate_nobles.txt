namespace = nfm_events_elevate_nobles

nfm_events_elevate_nobles.1 = {
    type = character_event
    title = nfm_events_elevate_nobles.1.t
    desc = {
        desc = nfm_events_elevate_nobles.1.d # Shared Intro
        desc = newline.desc
        # House-specific flavor based on selected type
        desc = nfm_events_elevate_nobles.1.d.house_desc
        # Family size flavor
        desc = newline.desc
        desc = nfm_events_elevate_nobles.1.d.familysize_desc
        # Family Type
        desc = newline.desc
        desc = nfm_events_elevate_nobles.1.d.housetype_desc
    }

    left_portrait = {
        character = root
        animation = personality_rational
    }

    right_portrait = {
        character = scope:new_house_head
    }

    immediate = {}

    theme = "administrative"
    override_background = {
        trigger = { always = yes }
        reference = courtyard
    }
    override_effect_2d = { reference = legend_glow }

    # Option 1: Grant them the Noble Title
    option = {
        name = nfm_events_elevate_nobles.1.a

        ## Create the Family Title
        custom_tooltip = {
            text = nfm_events_elevate_nobles.1.tt.becomes_head
            scope:new_house_head = {
                create_noble_family_effect = { GOVERNMENT_GIVER = root }
            }
        }

        ## Tooltip for House Modifier
        show_as_tooltip = {
            scope:new_house_head.house ?= { nfm_effect_add_house_modifier = yes }
        }

        ## Vassalize them since creating a new house doesn't anymore apparently
        nfm_effect_vassalize = {
            LIEGE = root
            VASSAL = scope:new_house_head
        }

        ## House Head Opinion Bonus and Merit
        # Split for Tooltip Reasons
        scope:new_house_head = {
            # House head gets larger bonus
            add_opinion = {
                target = root
                modifier = nfm_op_elevated
            }
            if = {
                limit = { root = { government_allows = merit } }
                hidden_effect = { nfm_effect_setup_merit_for_member = yes }
            }
        }

        ## Grant Merit and Opinion Bonuses
        scope:new_house_head.house = {
            every_house_member = {
                custom = nfm_loc_custom.every_house_member
                limit = { scope:new_house_head != this }

                add_opinion = {
                    target = root
                    modifier = nfm_op_family_elevated
                }

                hidden_effect = {
                    set_employer = scope:new_house_head
                }

                # Set up Merit and Examinations
                if = {
                    limit = { root = { government_allows = merit } }
                    hidden_effect = { nfm_effect_setup_merit_for_member = yes }
                    custom_tooltip = nfm_events_elevate_nobles.1.tt.gain_merit
                }
            }
        }

        ## Notify the Player
        send_interface_toast = {
            type = event_generic_good_with_text

            title = nfm_toast_elevated
            desc = nfm_toast_elevated_desc
            tooltip = nfm_toast_elevated_tt

            left_icon = scope:new_house_head
            right_icon = scope:new_house_head.house
        }
    }

    # Option 2: They aren't needed - Partial Refund
    option = {
        name = nfm_events_elevate_nobles.1.b

        ## Send them to the retirement fields (totally not just killing them)
        custom_tooltip = {
            text = nfm_events_elevate_nobles.1.b.flavor

            scope:new_house_head.house = {
                every_house_member = {
                    # Their spouse if they exist, because they're not house members
                    primary_spouse ?= {
                        if = {
                            limit = {
                                is_lowborn = yes
                            }
                            death = {
                                death_reason = death_disappearance
                            }
                        }
                    }
                    # Poof
                    death = {
                        death_reason = death_disappearance
                    }
                }
            }
        }

        ## Partial Cost Refunds (50%)
        # Prestige Cost
        add_prestige_no_experience = {
            value = nfm_value_elevate_cost_prestige
            multiply = 0.5
        }

        # Gold cost
        if = {
            limit = { has_treasury = no }
            add_gold = {
                value = nfm_value_elevate_cost_gold
                multiply = 0.5
            }
        }
        # Treasury cost
        if = {
            limit = { has_treasury = yes }
            add_treasury = {
                value = nfm_value_elevate_cost_treasury
                multiply = 0.5
            }
        }

        # Influence Cost
        change_influence_no_experience = { 
            value = nfm_value_elevate_cost_influence
            multiply = 0.5
        }

        ## Remove the Decision Cooldown
        remove_decision_cooldown = nfm_decision_elevate_nobles
    }
}
