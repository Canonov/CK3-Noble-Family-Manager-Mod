
# Elevate a Noble Family
nfm_decision_elevate_nobles = {
    title = {
        first_valid = {
            triggered_desc = {
                trigger = { culture = { has_cultural_pillar = heritage_japonic } }
                desc = nfm_decision_elevate_nobles_japanese
            }
            desc = nfm_decision_elevate_nobles
        }
    }

    desc = nfm_decision_elevate_nobles_desc
    selection_tooltip = nfm_decision_elevate_nobles_tooltip

    picture = {
        trigger = { culture = { has_graphical_east_asia_culture_group_trigger = yes } }
        reference = "gfx/interface/illustrations/decisions/tgp_scholars.dds"
    }
    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_recruitment.dds"
    }

    decision_group_type = courtier

    is_shown = {
        # for now, maybe
        is_ai = no

        # Must be at least a King
        is_landed = yes
        highest_held_title_tier >= tier_kingdom

        # Check Government
        government_allows = administrative
    }

    is_valid = {
    }

    widget = {
        gui = "decision_view_widget_option_list_generic"
        controller = decision_option_list_controller
        decision_to_second_step_button = "nfm_decision_elevate_nobles_confirm"
        show_from_start = yes # No would make it open the list after

        # House of Administrators
        item = {
            value = nfm_house_steward

            localization = nfm_house_steward_name
            current_description = nfm_house_steward_desc

            icon = "gfx/interface/icons/icon_skill_stewardship.dds"
            is_default = yes

            ai_chance = {
                value = 0
            }
        }

        # House of Warriors
        item = {
            value = nfm_house_martial

            localization = {
                first_valid = {
                    # Japanese - Samurai
                    triggered_desc = {
                        trigger = { culture = { has_cultural_pillar = heritage_japonic } }
                        desc = nfm_house_martial_name_samurai
                    }
                    desc = nfm_house_martial_name
                }
            }
            current_description = nfm_house_martial_desc

            icon = "gfx/interface/icons/icon_knight_acclaimed.dds"

            ai_chance = {
                value = 0
            }
        }

        # House of Scholars
        item = {
            value = nfm_house_learning

            localization = {
                first_valid = {
                    # Chinese - Confucian Scholars
                    triggered_desc = {
                        trigger = {
                            OR = {
                                culture = { has_cultural_tradition = tradition_scholar_officials }
                                faith.religion = religion:confucianism_religion
                            }
                        }
                        desc = nfm_house_learning_name_confucian
                    }
                    desc = nfm_house_learning_name
                }
            }
            current_description = {
                desc = nfm_house_learning_desc
            }

            icon = "gfx/interface/icons/icon_skill_learning.dds"

            ai_chance = {
                value = 0
            }
        }

        # House of Spies
        item = {
            value = nfm_house_intrigue

            localization = nfm_house_intrigue_name
            current_description = nfm_house_intrigue_desc

            icon = "gfx/interface/icons/icon_scheme.dds"

            ai_chance = {
                value = 0
            }
        }

        # House of Foreigners
        item = {
            value = nfm_house_foreign

            localization = nfm_house_foreign_name
            current_description = nfm_house_foreign_desc
            is_valid = {
                any_realm_county = { culture != root.culture }
            }

            icon = "gfx/interface/icons/icon_culture.dds"

            ai_chance = {
                value = 0
            }
        }

    }

    effect = {
        ## Choose a House Size
        random_list = {
            20 = {
                save_scope_value_as  = {
                    name = family_size
                    value = nfm_familysize_small
                }
                custom_tooltip = nfm_decision_elevate_nobles_small_tt
            }
            60 = {
                save_scope_value_as  = {
                    name = family_size
                    value = nfm_familysize_medium
                }
                custom_tooltip = nfm_decision_elevate_nobles_medium_tt
            }
            20 = {
                save_scope_value_as  = {
                    name = family_size
                    value = nfm_familysize_large
                }
                custom_tooltip = nfm_decision_elevate_nobles_large_tt
            }
        }

        ## Create the Family
        if = {
            limit = { scope:nfm_house_steward = yes }
            hidden_effect = {
                nfm_effect_create_house_steward = yes
            }
        }
        # Else the others, stewardship just to test

        # Create the Noble Family and Set up opinion and merit stuff
        hidden_effect = {
            scope:new_house_head ?= {
                create_noble_family_effect = { GOVERNMENT_GIVER = root }

                # House head gets larger bonus
                add_opinion = {
                    target = root
                    modifier = nfm_op_elevated
                }
            }

            scope:new_house_head.house ?= {
                every_house_member = {
                    # Other members get lesser opinion bonus
                    if = {
                        limit = { scope:new_house_head != this }
                        add_opinion = {
                            target = root
                            modifier = nfm_op_family_elevated
                        }
                    }

                    # Set up Merit and Examinations
                    if = {
                        limit = { root = { government_allows = merit } }
                        nfm_effect_setup_merit_for_member = yes
                    }
                }
            }
        }


        ## Notify the Player
        send_interface_toast = {
            type = event_generic_good_with_text

            title = nfm_toast_elevated
            desc = nfm_toast_elevated_desc
            tooltip = nfm_toast_elevated_tt

            left_icon = scope:new_house_head
            right_icon = scope:new_house_head.house
        }

    }

    cost = {
        prestige = {
            value = 200
        }

        # none if you have a treasury, 125 if not
        gold = {
            value = 0

            if = {
                limit = { has_treasury = no }
                add = medium_gold_value # was 125
            }

            max = 250
        }

        treasury = {
            value = 0

            if = {
                limit = { has_treasury = yes }
                add = medium_treasury_value # was 200
            }

            max = 500
        }

        influence = {
            # Will be given to them
            value = 400
        }
    }

    # Prevent AI from taking this decision
    ai_potential = {
        always = no
    }

    ai_check_interval = 0
}
