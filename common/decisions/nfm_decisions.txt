
# Elevate a Noble Family
nfm_decision_elevate_local_nobles = {
    title = nfm_decision_elevate_local_nobles
    desc = nfm_decision_elevate_local_nobles_desc
    selection_tooltip = nfm_decision_elevate_local_nobles_tooltip

    picture = {
        reference = "gfx/interface/illustrations/decisions/decision_dynasty_house.dds"
    }

    decision_group_type = courtier

    is_shown = {
        # for now, maybe
        is_ai = no

        # Must be at least a King
        is_landed = yes
        highest_held_title_tier >= tier_kingdom

        # Check Government
        government_allows = administrative
    }

    is_valid = {
    }

    widget = {
        gui = "decision_view_widget_option_list_generic"
        controller = decision_option_list_controller
        decision_to_second_step_button = "nfm_decision_elevate_local_nobles_confirm"
        show_from_start = yes

        # House of Administrators
        item = {
            value = nfm_house_steward
            is_shown = {
                always = yes
            }

            localization = nfm_house_steward_name
            current_description = nfm_house_steward_desc

            icon = "gfx/interface/icons/icon_skill_stewardship.dds"
            is_default = yes

            ai_chance = {
                value = 0
            }
        }
    
        # House of Warriors
        item = {
            value = nfm_house_martial
            is_shown = {
                always = yes
            }

            localization = nfm_house_martial_name
            current_description = nfm_house_martial_desc

            icon = "gfx/interface/icons/icon_knight_acclaimed.dds"

            ai_chance = {
                value = 0
            }
        }

        # House of Scholars
        item = {
            value = nfm_house_learning
            is_shown = {
                always = yes
            }

            localization = nfm_house_learning_name
            current_description = nfm_house_learning_desc

            icon = "gfx/interface/icons/icon_game_concept_innovation_fascination.dds"

            ai_chance = {
                value = 0
            }
        }

        # House of Spies
        item = {
            value = nfm_house_intrigue
            is_shown = {
                always = yes
            }

            localization = nfm_house_intrigue_name
            current_description = nfm_house_intrigue_desc

            icon = "gfx/interface/icons/icon_scheme.dds"

            ai_chance = {
                value = 0
            }
        }

        # House of Foreigners
        item = {
            value = nfm_house_foreign
            is_shown = {
                always = yes
            }
            is_valid = {
                any_realm_county = { culture != root.culture }
            }

            localization = nfm_house_foreign_name
            current_description = nfm_house_foreign_desc

            icon = "gfx/interface/icons/icon_culture.dds"

            ai_chance = {
                value = 0
            }
        }

    }

    effect = {
        ## Choose a House Size
        random_list = {
            20 = {
                save_scope_value_as  = {
                    name = family_size
                    value = nfm_familysize_small
                }
                custom_tooltip = nfm_decision_elevate_local_nobles_familywillbesmall_tt
            }
            60 = {
                save_scope_value_as  = {
                    name = family_size
                    value = nfm_familysize_medium
                }
                custom_tooltip = nfm_decision_elevate_local_nobles_familywillbemedium_tt
            }
            20 = {
                save_scope_value_as  = {
                    name = family_size
                    value = nfm_familysize_large
                }
                custom_tooltip = nfm_decision_elevate_local_nobles_familywillbelarge_tt
            }
        }

        ## Create the Family
        if = {
            limit = { scope:nfm_house_steward = yes }
            hidden_effect = {
                nfm_script_create_family_stewards = yes
            }  
        }
        # Else the others, stewardship just to test

        # Create the Noble House and add Opinion Modifiers for the new family
        hidden_effect = {
            scope:new_house_head ?= {
                create_noble_family_effect = { GOVERNMENT_GIVER = root }

                # House head gets larger bonus
                add_opinion = {
                    target = root
                    modifier = nfm_op_elevated
                }
            }

            scope:new_house_head.house ?= {
                every_house_member = {
                    limit = { scope:new_house_head != this }
                    
                    # Other members get lesser bonus
                    add_opinion = {
                        target = root
                        modifier = nfm_op_family_elevated
                    }
                }
            }
        }


        ## Notify the Player
        send_interface_toast = {
            type = event_generic_good_with_text

            title = nfm_toast_familyelevated
            desc = nfm_toast_familyelevated_desc
            tooltip = nfm_toast_familyelevated_tt

            left_icon = scope:new_house_head
            right_icon = scope:new_house_head.house
        }
        
    }

    # Prevent AI from taking this decision
    ai_potential = {
        always = no
    }

    ai_check_interval = 0
}
