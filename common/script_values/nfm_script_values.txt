# Elevate Noble Family - Cooldown Time (Days)
nfm_value_elevate_nobles_cooldown_days = {
    value = 365

    if = {
        limit = {
            has_variable = nfm_recently_founded_administration
        }
        value = 30
    }
    else_if = {
        limit = {
            any_noble_family = { count <= 5 }
        }
        multiply = 0.25
    }
    else_if = {
        limit = {
            any_noble_family = { count <= 10 }
        }
        multiply = 0.5
    }

    round = yes
}

# Get Tyranny amount for the Destroy House Decision
nfm_value_revoke_house_tyranny_gain = {
    value = 0

    if = {
        limit = { scope:sack_estate = yes }
        add = 7
    }
    if = {
        limit = { scope:eradicate_family = yes }
        add = 15
    }
    else_if = {
        limit = { scope:banish_family = yes }
        add = 10
    }

    # 50% cheaper if you have the legacy
    if = {
        limit = {
            dynasty ?= { has_dynasty_perk = ep3_administrative_legacy_5 }
        }
        multiply = 0.5
    }
}
