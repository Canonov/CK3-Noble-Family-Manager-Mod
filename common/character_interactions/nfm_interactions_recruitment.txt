################################
# Invite Foreign House Head
################################
nfm_int_invite_foreign_house_head = {
    category = interaction_category_vassal
    common_interaction = yes
    icon = invite_foreign_house_head

    desc = nfm_int_invite_foreign_house_head_desc

    greeting = positive

    interface_priority = 55

    use_diplomatic_range = yes

    is_shown = {
        scope:actor = {
            # Landed Player only
            is_ai = no
            is_landed = yes

            # Must be at least a King
            highest_held_title_tier >= tier_kingdom

            # Check Government
            government_allows = noble_families
        }
        scope:recipient = {
            nfm_trigger_is_foreign_landless_house_head = yes
        }

        NOT = { scope:actor = scope:recipient }
    }

    is_valid_showing_failures_only = {
        scope:actor = {
            is_available_quick = {
                imprisoned = no
                activity = no
                travel = no
                at_war = no
            }
        }

        scope:recipient = {
            # Not already at actor's court
            NOR = {
                is_courtier_of = scope:actor
                is_pool_guest_of = scope:actor
                host ?= scope:actor
            }

            # No house members already landed in your or their realm (Maybe merge these?)
            custom_tooltip = {
                text = nfm_int_invite_house_not_landed_your_realm_tt
                NOT = {
                    house = {
                        any_house_member = {
                            top_liege = scope:actor.top_liege
                            is_landed = yes
                        }
                    }
                }
            }
            custom_tooltip = {
                text = nfm_int_invite_house_not_landed_their_realm_tt
                NOT = {
                    house = {
                        any_house_member = {
                            top_liege = scope:recipient.top_liege
                            is_landed = yes
                        }
                    }
                }
            }

            # Not doing travelling, imprisoned, or otherwise unavailable
            is_available_quick = {
                travel = no
                imprisoned = no
            }
        }
    }

    is_valid = {
        # Only hard block for state faith that considers recipient's faith "evil"
        # Other faith conflicts are handled via AI acceptance modifiers
        trigger_if = {
            limit = {
                scope:actor = {
                    government_allows = state_faith
                }
            }
            scope:recipient.faith = {
                faith_hostility_level = {
                    target = scope:actor.primary_title.state_faith
                    value >= faith_evil_level
                }
            }
        }
    }

    # Send options
    send_options_exclusive = no

    # Option: Offer generous compensation (increases acceptance)
    send_option = {
        flag = generous_offer
        localization = nfm_int_invite_generous_offer

        is_shown = {
            always = yes
        }

        current_description = nfm_int_invite_generous_offer_desc
    }

    # Option: Use a hook
    send_option = {
        is_valid = {
            scope:actor = { has_usable_hook = scope:recipient }
        }
        flag = hook
        localization = GENERIC_SPEND_A_HOOK
    }

    should_use_extra_icon = {
        scope:actor = { has_usable_hook = scope:recipient }
    }
    extra_icon = "gfx/interface/icons/character_interactions/hook_icon.dds"

    cost = {
        gold = {
            value = nfm_value_invite_house_gold_cost

            # Generous offer doubles it
            if = {
                limit = { scope:generous_offer = yes }
                multiply = 2
            }
        }

        prestige = 250

        influence = { value = nfm_value_invite_house_influence_cost}
    }

    ai_accept = {
        # ==========================================
        # BASE & OPINION
        # ==========================================

        base = -30

        # Opinion of actor
        opinion_modifier = {
            opinion_target = scope:actor
            multiplier = 0.5
            step = 5
            max = 25
        }

        # Negative opinion of current liege makes them more willing to leave
        opinion_modifier = {
            trigger = {
                exists = liege
            }
            opinion_target = liege
            multiplier = -0.3
            step = 5
            max = 15
        }

        # ==========================================
        # OFFER MODIFIERS
        # ==========================================

        # House head bonus - they're being recruited as a house head (Freebie for flavour)
        modifier = {
            add = 5
            desc = nfm_ai_will_retain_noble_family
        }

        # Generous offer bonus
        modifier = {
            scope:generous_offer = yes
            add = {
                value = 30

                if = {
                    limit = { has_trait = greedy }
                    add = 20
                }
            }

            desc = nfm_ai_generous_offer
        }

        # True Ruler perk boost
        modifier = {
            scope:actor = { has_perk = true_ruler_perk }
            add = true_ruler_value
            desc = offer_vassalization_true_ruler_perk_tt
        }

        # HOOK
        modifier = {
            add = 40
            scope:hook = yes
            scope:actor = { NOT = { has_strong_hook = scope:recipient } }
            desc = SCHEME_WEAK_HOOK_USED
        }

        # STRONG HOOK
        modifier = {
            add = 100
            scope:hook = yes
            scope:actor = { has_strong_hook = scope:recipient }
            desc = SCHEME_STRONG_HOOK_USED
        }


        # ==========================================
        # RANK MODIFIERS
        # ==========================================

        # Actor is higher/lower tier ruler
        modifier = {
            add = 10
            scope:actor = {
                tier_difference = {
                    target = scope:recipient.top_liege
                    value = 1
                }
            }
            desc = nfm_ai_liege_rank_diff
        }
        modifier = {
            add = 20
            scope:actor = {
                tier_difference = {
                    target = scope:recipient.top_liege
                    value = 2
                }
            }
            desc = nfm_ai_liege_rank_diff
        }

        modifier = {
            add = -10
            scope:actor = {
                tier_difference = {
                    target = scope:recipient.top_liege
                    value = -1
                }
            }
            desc = nfm_ai_liege_rank_diff
        }
        modifier = {
            add = -20
            scope:actor = {
                tier_difference = {
                    target = scope:recipient.top_liege
                    value = -2
                }
            }
            desc = nfm_ai_liege_rank_diff
        }

        # ==========================================
        # CULTURE MODIFIERS
        # ==========================================

        # Cultural acceptance penalties (tiered)
        modifier = {
            culture = {
                NOT = { this = scope:actor.culture }
                cultural_acceptance = { target = scope:actor.culture value < 75 }
                cultural_acceptance = { target = scope:actor.culture value >= 50 }
            }
            add = -5
            desc = HOSTAGE_CULTURAL_ACCEPTANCE_REASON
        }

        modifier = {
            culture = {
                NOT = { this = scope:actor.culture }
                cultural_acceptance = { target = scope:actor.culture value < 50 }
                cultural_acceptance = { target = scope:actor.culture value >= 25 }
            }
            add = -10
            desc = HOSTAGE_CULTURAL_ACCEPTANCE_REASON
        }

        modifier = {
            culture = {
                NOT = { this = scope:actor.culture }
                cultural_acceptance = { target = scope:actor.culture value < 25 }
            }
            add = -15
            desc = HOSTAGE_CULTURAL_ACCEPTANCE_REASON
        }

        # ==========================================
        # FAITH MODIFIERS
        # ==========================================

        # Faith hostility toward actor (tiered)
        modifier = {
            faith = {
                NOT = { this = scope:actor.faith }
                faith_hostility_level = {
                    target = scope:actor.faith
                    value = faith_evil_level
                }
            }
            add = -15
            desc = HOSTAGE_FAITH_EVIL
        }

        modifier = {
            faith = {
                NOT = { this = scope:actor.faith }
                faith_hostility_level = {
                    target = scope:actor.faith
                    value = faith_hostile_level
                }
            }
            add = -10
            desc = HOSTAGE_FAITH_HOSTILE
        }

        modifier = {
            faith = {
                NOT = { this = scope:actor.faith }
                faith_hostility_level = {
                    target = scope:actor.faith
                    value = faith_astray_level
                }
            }
            add = -5
            desc = HOSTAGE_FAITH_ASTRAY
        }

        # Faith comparative - stay factor (liege shares, actor doesn't)
        modifier = {
            faith = liege.faith
            faith != scope:actor.faith
            add = -5
            desc = nfm_ai_liege_shares_faith
        }

        # ==========================================
        # PRESTIGE & STATUS MODIFIERS
        # ==========================================

        # Their house prestige - higher prestige houses expect better offers
        modifier = {
            dynasty.dynasty_prestige_level >= 3
            add = -20
            desc = nfm_ai_high_prestige_house
        }

        modifier = {
            dynasty.dynasty_prestige_level >= 5
            add = -20  # Additional penalty for illustrious houses
            desc = nfm_ai_high_prestige_house
        }

        # Liege at war - don't leave during crisis
        modifier = {
            liege = { is_at_war = yes }
            add = -30
            desc = nfm_ai_liege_at_war
        }

        # Recipient at war themselves - NEVER accept
        modifier = {
            is_at_war = yes
            add = -1000
            desc = nfm_ai_at_war
        }

        # Disloyalty concerns - unless ambitious or already dislikes liege
        modifier = {
            NOR = {
                has_trait = ambitious
                has_trait = disloyal
                opinion = {
                    target = liege
                    value < -35
                }
            }
            add = -10
            desc = nfm_ai_seen_as_disloyal
        }

        # Liege is more prestigious - stay factor
        modifier = {
            liege.prestige_level > scope:actor.prestige_level
            add = -15
            desc = nfm_ai_liege_more_prestigious
        }

        # Actor is more prestigious - pull factor
        modifier = {
            scope:actor.prestige_level > liege.prestige_level
            add = 10
            desc = nfm_ai_actor_more_prestigious
        }

        # ==========================================
        # RELATION MODIFIERS
        # ==========================================

        # Same dynasty connection
        modifier = {
            dynasty = scope:actor.dynasty
            house != scope:actor.house
            add = 20
            desc = AI_SAME_DYNASTY_REASON
        }

        # Same house - even stronger
        modifier = {
            house = scope:actor.house
            add = 30
            desc = AI_SAME_HOUSE_REASON
        }

        # Family ties with actor
        modifier = {
            is_close_family_of = scope:actor
            add = 15
            desc = AI_YOUR_RELATIVE
        }

        modifier = {
            is_extended_family_of = scope:actor
            NOT = { is_close_family_of = scope:actor }
            add = 5
            desc = AI_YOUR_RELATIVE
        }

        # Rivalry with actor
        modifier = {
            has_relation_rival = scope:actor
            add = -200
            desc = AI_YOUR_RIVAL
        }

        # Friendship with actor
        modifier = {
            has_relation_friend = scope:actor
            add = 20
            desc = AI_YOUR_FRIEND
        }

        # Lover
        modifier = {
            has_relation_lover = scope:actor
            add = 20
            desc = AI_YOUR_LOVER
        }

        # ==========================================
        # PERSONALITY & SITUATION MODIFIERS
        # ==========================================

        # Liege is arbitrary (and actor isn't) - bonus to leave
        modifier = {
            liege = { has_trait = arbitrary }
            NOT = { scope:actor = { has_trait = arbitrary } }
            add = 15
            desc = nfm_ai_liege_arbitrary
        }

        # Content trait
        modifier = {
            has_trait = content
            add = -25
            desc = INTERACTION_CONTENT
        }

        # Ambitious trait
        modifier = {
            has_trait = ambitious
            add = 15
            desc = INTERACTION_AMBITIOUS
        }

        # Recipient is just - values honorable dealings
        modifier = {
            has_trait = just
            add = 10
            desc = nfm_ai_recipient_just
        }

        # Both are just - shared values bonus
        modifier = {
            has_trait = just
            scope:actor = { has_trait = just }
            add = 10
            desc = nfm_ai_both_just
        }

        # Liege has tyranny - push factor (three tiers: 10/20/40)
        modifier = {
            liege.tyranny >= 10

            add = {
                value = 10

                if = {
                    limit = { liege.tyranny >= 20 }
                    add = 10
                }
                if = {
                    limit = { liege.tyranny >= 40 }
                    add = 20
                }
            }
            desc = nfm_ai_liege_tyrannical
        }

        # Actor is tyrannical - deterrent (three tiers: 10/20/40)
        modifier = {
            scope:actor.tyranny >= 10

            add = {
                value = -10

                if = {
                    limit = { scope:actor.tyranny >= 20 }
                    add = -10
                }
                if = {
                    limit = { scope:actor.tyranny >= 40 }
                    add = -20
                }
            }
            desc = nfm_ai_actor_tyrannical
        }

        # Escaping punishment - liege has imprisonment reason and dislikes them
        modifier = {
            liege = {
                has_imprisonment_reason = prev
                opinion = {
                    target = prev
                    value < 0
                }
            }
            add = 25
            desc = nfm_ai_escaping_punishment
        }

        # Additional for execution/banishment reasons
        modifier = {
            liege = {
                OR = {
                    has_execute_reason = prev
                    has_banish_reason = prev
                }
            }
            add = 15
            desc = nfm_ai_escaping_punishment
        }

        # Legitimacy - Actor has higher legitimacy
        modifier = {
            scope:actor.legitimacy_level > liege.legitimacy_level
            add = 10
            desc = nfm_ai_actor_more_legitimate
        }

        # Legitimacy - Liege has higher legitimacy (stay factor)
        modifier = {
            liege.legitimacy_level > scope:actor.legitimacy_level
            add = -10
            desc = nfm_ai_liege_more_legitimate
        }

        # Loyal trait - major penalty to leaving
        modifier = {
            has_trait = loyal
            add = -35
            desc = nfm_ai_loyal_to_liege
        }

        # Disloyal trait - bonus to leaving
        modifier = {
            has_trait = disloyal
            add = 10
            desc = nfm_ai_disloyal
        }

        # ==========================================
        # HOUSE RELATION MODIFIERS
        # ==========================================

        # Positive house relation with actor's house
        modifier = {
            house ?= {
                has_house_relation_with = scope:actor.house
                any_house_relation = {
                    any_relation_house = { this = scope:actor.house }
                    OR = {
                        has_house_relation_level = cordial
                        has_house_relation_level = friendly
                        has_house_relation_level = amity
                    }
                }
            }
            add = 15
            desc = nfm_ai_house_relation
        }

        # Negative house relation with actor's house
        modifier = {
            house ?= {
                has_house_relation_with = scope:actor.house
                any_house_relation = {
                    any_relation_house = { this = scope:actor.house }
                    OR = {
                        has_house_relation_level = quarrel
                        has_house_relation_level = rivalry
                        has_house_relation_level = feud
                    }
                }
            }
            add = -20
            desc = nfm_ai_house_relation
        }

        # ==========================================
        # DEBT MODIFIERS
        # ==========================================

        # Liege in debt - unstable realm
        modifier = {
            liege = { is_in_debt = yes }
            add = 10
            desc = nfm_ai_liege_in_debt
        }

        # Liege in severe debt (debt level 2+)
        modifier = {
            liege = { debt_level >= 2 }
            add = 10
            desc = nfm_ai_liege_in_debt
        }

        # Treasury-specific debt (if liege uses treasury system)
        modifier = {
            liege = {
                has_treasury = yes
                treasury_debt_level >= 1
            }
            add = 10
            desc = nfm_ai_liege_treasury_debt
        }
    }

    on_accept = {
        scope:recipient.liege = { save_scope_as = old_liege }
        scope:recipient.primary_title = { save_scope_as = old_nf_title }

        # Add them as courtier and destroy their old title
        hidden_effect = {
            scope:recipient = { destroy_title = primary_title }
            scope:actor = { add_courtier = scope:recipient }
        }

        scope:recipient = {
            # Create new family and vassalize them
            custom_tooltip = {
                text = nfm_int_invite_house_head_move_noble_family_tt
                create_noble_family_effect = { GOVERNMENT_GIVER = scope:actor }

                # Copy title history from old title to new title
                scope:new_title = { copy_title_history = scope:old_nf_title }

                nfm_effect_vassalize = {
                    LIEGE = scope:actor
                    VASSAL = scope:recipient
                }
            }

            add_character_modifier = {
                modifier = nfm_modifier_house_head_poached_disloyalty
                years = 15
            }

            # Grant them resources
            add_prestige = 200
            change_influence = nfm_value_invite_house_influence_grant
        }

        # Pay gold to recipient
        scope:actor = {
            show_as_tooltip = {
                pay_short_term_gold = {
                    target = scope:recipient
                    gold = {
                        value = nfm_value_invite_house_gold_cost
                        if = {
                            limit = { scope:generous_offer = yes }
                            multiply = 2
                        }
                    }
                }
            }
            hidden_effect = {
                scope:recipient = {
                    add_short_term_gold = {
                        value = nfm_value_invite_house_gold_cost
                        if = {
                            limit = { scope:generous_offer = yes }
                            multiply = 2
                        }
                    }
                }
            }
        }

        # Notify former liege
        scope:old_liege = {
            send_interface_message = {
                type = event_generic_neutral
                title = nfm_msg_house_head_left_realm
                right_icon = scope:recipient

                custom_tooltip = nfm_msg_house_head_left_realm_desc

                # Opinion penalty toward the poacher
                add_opinion = {
                    modifier = nfm_op_poached_house_head
                    target = scope:actor
                }

                # Opinion penalty toward the traitorous vassal
                add_opinion = {
                    modifier = nfm_op_abandoned_liege
                    target = scope:recipient
                }
            }
        }

        # House Relations
        # Improve relation with invited house
        if = {
            limit = {
                scope:actor.house != scope:recipient.house
            }
            scope:actor.house = {
                change_house_relation_effect = {
                    HOUSE = scope:recipient.house
                    VALUE = house_relation_improve_medium_value
                    REASON = nfm_invited_house
                    CHAR = scope:actor
                    TARGET_CHAR = scope:recipient
                    TITLE = scope:new_title
                }
            }
        }

        # Notification & Stress
        scope:actor = {
            send_interface_message = {
                type = event_generic_good
                title = nfm_msg_invited_house_head
                desc = nfm_msg_invited_house_head_desc
                right_icon = scope:recipient
            }

            stress_impact = {
                shy = minor_stress_impact_gain
                paranoid = minor_stress_impact_gain
                gregarious = minor_stress_impact_loss
                ambitious = minor_stress_impact_loss
            }
        }
    }

    on_decline = {
        scope:actor = {
            send_interface_message = {
                type = event_generic_bad
                title = nfm_msg_invite_declined
                right_icon = scope:recipient

                custom_tooltip = nfm_msg_invite_declined_desc
            }
        }
    }

    ai_will_do = {
        base = 0
    }
}
