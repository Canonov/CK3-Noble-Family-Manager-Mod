## nfm_effect_vassalize
# Vassalize $VASSAL$ into $LIEGE$
#
# @input $LIEGE$ - The Character who is vassalizing them.
# @input $VASSAL$ - The Character who is vassalized
nfm_effect_vassalize = {
    $LIEGE$ = { save_scope_as = new_liege }
    $VASSAL$ = { save_scope_as = new_vassal }

    create_title_and_vassal_change = {
        type = swear_fealty
        save_scope_as = change
    }
    scope:new_vassal = {
        change_liege = {
            liege = scope:new_liege
            change = scope:change
        }
    }
    resolve_title_and_vassal_change = scope:change
}

## nfm_effect_elevate_lowborn
# Elevates a lowborn character to nobility by creating a temporary title that
# generates a dynasty, then destroys it. The target gains prestige and an opinion bonus.
#
# @input $GIVER$ - The ruler granting nobility
# @input $TARGET$ - The lowborn character being elevated
nfm_effect_elevate_lowborn = {
    $GIVER$ = { save_scope_as = giver }
    $TARGET$ = { save_scope_as = target }

    hidden_effect = {
        scope:target = {
            # Create and Destroy a Dynamic Title to generate a dynasty
            create_dynamic_title = {
                tier = duchy
                name = nfm_lowborn_grant_title_name
            }
            create_title_and_vassal_change = {
                type = created
                save_scope_as = change
                add_claim_on_loss = no
            }
            scope:new_title = {
                set_capital_county = scope:giver.capital_county
                set_landless_title = yes
                set_destroy_on_gain_same_tier = yes
                set_no_automatic_claims = yes
                set_can_be_named_after_dynasty = no
                change_title_holder = {
                    holder = prev
                    change = scope:change
                }
                set_coa = scope:giver.primary_title
            }
            resolve_title_and_vassal_change = scope:change
            destroy_title = scope:new_title
        }

        scope:giver = {
            add_courtier = scope:target
        }
    }

    scope:target = {
        custom_tooltip = nfm_elevate_lowborn_tt
        add_opinion = {
            target = scope:giver
            modifier = nfm_op_granted_nobility
        }
        add_prestige = 100
    }
}
