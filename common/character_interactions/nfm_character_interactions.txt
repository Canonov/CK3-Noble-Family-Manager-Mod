nfm_int_grant_house = {
    category = interaction_category_vassal
    common_interaction = no
    icon = icon_shift_privileges

    desc = nfm_int_grant_house_desc

    greeting = positive

    auto_accept = yes

    is_shown = {
        scope:actor = {
            # for now, maybe
            is_ai = no

            is_landed = yes

            # Must be at least a King
            highest_held_title_tier >= tier_kingdom

            # Check Government
            government_allows = administrative
        }
        scope:recipient = {
            is_courtier_of = scope:actor
            is_ruler = no
            is_lowborn = no
            is_adult = yes
            is_imprisoned = no

            # Check Marriage Type
            OR = {
                # are unmarried
                is_married = no
                # dominant partner in at least one of their marriages
                AND = {
                    is_female = yes
                    matrilinear_marriage = yes
                }
                AND = {
                    is_male = yes
                    matrilinear_marriage = no
                }
                # Your marriage is same-sex
                NOR = {
                    AND = {
                        any_consort = {
                            is_female = yes
                        }
                        is_male = yes
                    }
                    AND = {
                        any_consort = {
                            is_male = yes
                        }
                        is_female = yes
                    }
                }
            }
        }
        NOT = { scope:actor = scope:recipient }
    }

    is_valid_showing_failures_only = {
        scope:recipient = {
            NOT = {
                has_trait = devoted
                has_trait = gallivanter
            }
            custom_description = {
                text = is_not_theocratic_court_chaplain
                subject = scope:recipient
                NAND = {
                    scope:actor.faith = {
                        has_doctrine = doctrine_theocracy_temporal
                    }
                    scope:recipient = {
                        faith = {
                            has_doctrine = doctrine_theocracy_temporal
                        }
                        has_council_position = councillor_court_chaplain
                    }
                    scope:actor.faith.religious_head != scope:recipient
                }
            }
        }
    }

    on_accept = {
        scope:recipient = {
            custom_tooltip = {
                text = create_noble_family_tt

                create_noble_family_effect = { GOVERNMENT_GIVER = scope:actor }
            }

            custom_tooltip = {
                text = nfm_int_grant_house_costs_tt

                add_prestige = 250
                change_influence = 150
                add_gold = 50
            }


            if = {
                limit = {
                    scope:actor = { government_allows = merit }
                }
                change_merit_level = 3
                add_character_modifier = {
                    modifier = tgp_passed_provincial_exam_modifier
                }
            }

            # Opinion boost towards the granter
            add_opinion = {
                modifier = nfm_op_elevated
                target = scope:actor
            }
        }
        scope:actor = {
            stress_impact = {
                paranoid = minor_stress_impact_gain
                greedy = minor_stress_impact_gain
                arrogant = minor_stress_impact_gain
                generous = minor_stress_impact_loss
                trusting = minor_stress_impact_loss
                peasant_leader = major_stress_impact_loss
                populist_leader = major_stress_impact_loss
            }

            # Grant Hook
            if = {
                limit = {
                    scope:recipient.dynasty = dynasty
                }
                # Grant strong loyalty hook to the granter
                add_hook = {
                    target = scope:recipient
                    type = loyalty_hook
                }
            }
            else = {
                # Grant hook to the granter
                add_hook = {
                    target = scope:recipient
                    type = obligation_hook
                }
            }
        }
    }

    cost = {
        prestige = {
            value = 250

            # Vassals pay double
            if = {
                limit = { scope:actor = { is_independent_ruler = no } }
                multiply = 2
            }
        }
        gold = {
            value = 75

            # Treasury Pays 50 Less
            if = {
                limit = {
                    scope:actor = { has_treasury = yes }
                }
                value = 25
            }
        }

        treasury = {
            value = 0

            # Treasury Pays 50
            if = {
                limit = {
                    scope:actor = { has_treasury = yes }
                }
                value = 50
            }
        }

        influence = {
            value = 150

            # Vassals pay double
            if = {
                limit = { scope:actor = { is_independent_ruler = no } }
                multiply = 2
            }
        }

    }
}

## Destroy Noble House
# TODO allow it to be done tyrannically and add a crown auth requirement
nfm_int_revoke_house = {
    icon = icon_governor_removal
    category = interaction_category_prison
    common_interaction = yes
    interface_priority = 66
    desc = nfm_int_revoke_house_desc
    options_heading = nfm_int_revoke_house_options

    greeting = negative
    send_options_exclusive = no

    is_shown = {
        scope:actor = {
            is_ai = no
            top_liege = this

            # Check Government
            OR = {
                government_allows = administrative
                has_government = japan_feudal_government
            }
        }

        scope:recipient = {
            top_liege ?= scope:actor
            is_ruler = yes
            is_imprisoned_by = scope:actor

            any_held_title = {
                is_noble_family_title = yes
            }
        }
    }

    is_highlighted = {
        OR = {
            scope:actor = { has_trait = sadistic }
            scope:actor = { has_relation_rival = scope:recipient }
            scope:actor = {
                OR = {
                    has_execute_reason = scope:recipient
                    has_banish_reason = scope:recipient
                }
            }
        }
    }

    highlighted_reason = HIGHLIGHTED_HAS_EXECUTION_REASON

    is_valid_showing_failures_only = {
        scope:actor = {
            is_imprisoned = no
            NOT = { exists = involved_activity }
            NOT = { is_allied_to = scope:recipient }
        }

        # They must not be a ceremonial liege and must not have a hook on you.
        scope:recipient = {
            NOT = { has_strong_hook = scope:actor }
            tgp_is_ceremonial_liege_trigger = no
        }

        scope:recipient.house = {
            custom_tooltip = {
                text = nfm_int_revoke_house_no_rulers_tt

                NOT = {
                    any_house_member = {
                        is_landed = yes
                        top_liege = scope:actor
                    }
                }

            }
        }
    }

    auto_accept = {
        custom_description = {
            text = nfm_int_revoke_house_auto_accept_reason
            always = yes
        }
    }

    on_auto_accept = {
        # covered by on_accept
    }

    # Option: Sack their Estates
    send_option = {
        localization = nfm_int_revoke_sack
        flag = sack_estate

        is_shown = {
            always = yes
        }

        # Tooltip
        current_description = nfm_int_revoke_sack_desc
    }

    # Option: Imprison their family
    send_option = {
        localization = nfm_int_revoke_imprison
        flag = imprison_family

        is_shown = {
            scope:recipient.house ?= {
                any_house_member = { count > 1 }
            }
        }

        is_valid = {
            # Incompatible
            custom_tooltip = {
                text = nfm_int_revoke_imprison_exclusive_tt
                scope:banish_family = no
            }


            # Their estate must also be sacked
            custom_tooltip = {
                text = nfm_int_revoke_imprison_requires_sack_tt

                scope:sack_estate = yes
            }

            custom_tooltip = {
                text = nfm_int_revoke_imprison_traits_tt

                scope:actor = {
                    OR = {
                        dynasty ?= { has_dynasty_perk = ep3_administrative_legacy_5 }
                        has_trait = vengeful
                        has_trait = sadistic
                        has_trait = arbitrary
                    }
                }
            }
        }

        # Tooltip
        current_description = nfm_int_revoke_imprison_desc
    }

    # Option: Banish their family
    send_option = {
        localization = nfm_int_revoke_banish
        flag = banish_family

        is_shown = {
            scope:recipient.house ?= {
                any_house_member = { count > 1 }
            }
        }

        is_valid = {
            # Incompatible
            custom_tooltip = {
                text = nfm_int_revoke_banish_exclusive_tt
                scope:imprison_family = no
            }
        }

        # Tooltip
        current_description = nfm_int_revoke_banish_desc
    }


    on_accept = {
        ## Setup
        # Save their family title
        scope:recipient = {
            random_held_title = {
                limit = { is_noble_family_title = yes }
                save_scope_as = family_title
            }
        }
        # Set up and Calculate Domicile worth
        scope:family_title.title_domicile = {
            save_scope_value_as = {
                name = domicile_value
                value = nfm_value_domicileworth
            }
        }

        # Categorize family into those_to_kill and those_to_banish
        scope:recipient = {
            every_close_or_extended_family_member = {
                limit = {
                    top_liege ?= scope:actor
                    house ?= scope:recipient.house

                    NOT = { is_close_family_of = scope:actor }
                }
                if = {
                    limit = {
                        age >= 16
                        age < 70
                        is_faith_dominant_gender = yes
                        OR = {
                            is_parent_of = scope:recipient
                            is_child_of = scope:recipient
                        }
                    }
                    add_to_list = those_to_kill
                }
                else = {
                    add_to_list = those_to_banish
                }
            }
            # Add spouses to banish list
            every_spouse = {
                limit = {
                    top_liege = scope:actor
                }
                add_to_list = those_to_banish
            }
        }


        ## Gold
        # Option: Sacking the Estate
        if = {
            limit = { scope:sack_estate = yes }

            # 50% the value as loot
            scope:actor = {
                add_gold = {
                    value = scope:domicile_value
                    multiply = 0.5
                }
            }
            # 20% the value recovered
            scope:recipient = {
                add_gold = {
                    value = scope:domicile_value
                    multiply = 0.2
                }
            }
        }
        else = {
            # 20% the value as loot
            scope:actor = {
                add_gold = {
                    value = scope:domicile_value
                    multiply = 0.2
                }
            }
            # 80% the value recovered
            scope:recipient = {
                add_gold = {
                    value = scope:domicile_value
                    multiply = 0.8
                }
            }
        }


        ## Destroy the Title
        scope:actor = { destroy_title = scope:family_title }


        ## Family Punishment Options
        # Banish their family (all members)
        if = {
            limit = { scope:banish_family = yes }

            # Banish those marked to banish
            every_in_list = {
                list = those_to_banish
                limit = {
                    scope:recipient != this
                }
                banish = yes
                add_opinion = {
                    modifier = banished_me
                    target = scope:actor
                }
            }
            # Also banish those marked to kill (since we're not imprisoning)
            every_in_list = {
                list = those_to_kill
                limit = {
                    scope:recipient != this
                }
                banish = yes
                add_opinion = {
                    modifier = banished_me
                    target = scope:actor
                }
            }
        }
        # Imprison/Execute their family
        else_if = {
            limit = { scope:imprison_family = yes }

            # Execute those marked to kill
            custom_tooltip = {
                text = nfm_int_revoke_kill_explanation_tt
            }
            every_in_list = {
                list = those_to_kill
                limit = {
                    scope:recipient != this
                    is_alive = yes
                }
                hidden_effect = {
                    death = {
                        death_reason = death_execution
                        killer = scope:actor
                    }
                }
            }

            scope:recipient = {
                death = {
                    death_reason = death_execution
                    killer = scope:actor
                }
            }

            # Banish those marked to banish
            custom_tooltip = {
                text = nfm_int_revoke_banish_explanation_tt
            }
            every_in_list = {
                list = those_to_banish
                limit = {
                    scope:recipient != this
                }
                banish = yes
                add_opinion = {
                    modifier = banished_me
                    target = scope:actor
                }
            }
        }


        ## Post Effects
        # Banish Them or keep them there
        scope:recipient = {
            if = {
                limit = { scope:imprison_family = no }

                # Banish them
                add_opinion = {
                    modifier = banished_me
                    target = scope:actor
                }

                # TODO maybe guarantee it and give them a boost
                # They might become an adventurer
                release_from_prison = yes
                banish_effect = { BANISHER = scope:actor }

                # If they are an adventurer afterwards.. EXPEL THEM
                if = {
                    limit = { is_landless_ruler = yes }
                    set_variable = {
                        name = adventurer_expeller
                        value = scope:actor
                        months = 4
                    }
                    scope:actor = { save_scope_as = expeller }
                    trigger_event = {
                        id = ep3_interactions_events.0500
                        days = 1
                    }
                    trigger_event = {
                        id = ep3_interactions_events.0501
                        months = 3
                    }
                }
            }
        }

        # Tyranny and Stress Gain
        scope:actor = {
            stress_impact = {
                # Stress Gain
                forgiving = minor_stress_impact_gain
                compassionate = minor_stress_impact_gain
                # Stress Loss
                vengeful = medium_stress_impact_loss
            }
            add_tyranny = nfm_value_familydestructiontyranny

            # Extra tyranny when no valid reason (2x total)
            if = {
                limit = {
                    NOR = {
                        has_revoke_title_reason = scope:recipient
                        has_banish_reason = scope:recipient
                        has_execute_reason = scope:recipient
                    }
                }
                custom_tooltip = {
                    text = nfm_int_revoke_no_reason_tyranny_tt
                    add_tyranny = nfm_value_familydestructiontyranny
                }
            }
        }

    }

    cost = {
        # 200-800, base of 100 and 50 per dynasty level
        prestige = {
            value = 100

            add = {
                value = 50
                multiply = scope:recipient.dynasty.dynasty_prestige_level
            }

            min = 200
            max = 800
        }

        # Match half their influence plus a base cost of 50, 100 if they don't have influence
        influence = {
            value = 50

            if = {
                limit = {
                    scope:recipient = { government_allows = administrative }
                }
                add = {
                    value = scope:recipient.influence
                    multiply = 0.5
                }
            }
            else = {
                add = 50
            }

            min = 50
            max = 600
        }
    }
}
