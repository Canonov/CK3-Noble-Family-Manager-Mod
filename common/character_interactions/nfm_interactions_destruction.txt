
## Destroy Noble House
nfm_int_revoke_house = {
    icon = icon_governor_removal
    category = interaction_category_prison
    common_interaction = yes
    interface_priority = 66
    desc = nfm_int_revoke_house_desc
    options_heading = nfm_int_revoke_house_options

    greeting = negative
    send_options_exclusive = no

    is_shown = {
        scope:actor = {
            is_ai = no
            top_liege = this

            # Check Government
            government_allows = noble_families
        }

        scope:recipient = {
            top_liege ?= scope:actor
            is_ruler = yes
            is_imprisoned_by = scope:actor

            any_held_title = {
                is_noble_family_title = yes
            }
        }
    }

    is_highlighted = {
        OR = {
            scope:actor = { has_trait = sadistic }
            scope:actor = { has_relation_rival = scope:recipient }
            scope:actor = {
                OR = {
                    has_execute_reason = scope:recipient
                    has_banish_reason = scope:recipient
                }
            }
        }
    }

    highlighted_reason = HIGHLIGHTED_HAS_EXECUTION_REASON

    is_valid = {
        # No Member of their house can have Land
        scope:recipient.house = {
            custom_description = {
                text = nfm_loc_house_must_be_unlanded
                NOT = {
                    any_house_member = {
                        is_landed = yes
                        top_liege = scope:actor
                    }
                }
            }
        }
        # Not allied to them
        scope:actor = {
            NOT = { is_allied_to = scope:recipient }
        }
        # Requires at least Limited Crown Authority (or Equivalent)
        scope:actor = {
            nfm_trigger_has_authority_for_destruction = yes
        }
    }

    is_valid_showing_failures_only = {
        # You must be available
        scope:actor = {
            is_available_quick = {
                at_war = no
                imprisoned = no
                travel = no
            }
        }

        # They must not be a ceremonial liege and must not have a hook on you.
        scope:recipient = {
            NOT = { has_strong_hook = scope:actor }
            tgp_is_ceremonial_liege_trigger = no
        }
    }

    auto_accept = {
        custom_description = {
            text = nfm_int_revoke_house_auto_accept_reason
            always = yes
        }
    }

    on_auto_accept = {
        # covered by on_accept
    }

    # Option: Sack their Estates
    send_option = {
        localization = nfm_int_revoke_sack
        flag = sack_estate

        is_shown = {
            always = yes
        }

        # Tooltip
        current_description = nfm_int_revoke_sack_desc
    }

    # Option: Eradicate their family
    send_option = {
        localization = nfm_int_revoke_eradicate
        flag = eradicate_family

        is_shown = {
            scope:recipient.house ?= {
                any_house_member = { count > 1 }
            }
        }

        is_valid = {
            # Incompatible
            custom_tooltip = {
                text = nfm_int_revoke_eradicate_exclusive_tt
                scope:banish_family = no
            }


            # Their estate must also be sacked
            custom_tooltip = {
                text = nfm_int_revoke_eradicate_requires_sack_tt

                scope:sack_estate = yes
            }

            custom_tooltip = {
                text = nfm_int_revoke_eradicate_traits_tt

                scope:actor = {
                    OR = {
                        dynasty ?= { has_dynasty_perk = ep3_administrative_legacy_5 }
                        has_trait = vengeful
                        has_trait = sadistic
                        has_trait = arbitrary
                    }
                }
            }
        }

        # Tooltip
        current_description = nfm_int_revoke_eradicate_desc
    }

    # Option: Banish their family
    send_option = {
        localization = nfm_int_revoke_banish
        flag = banish_family

        is_shown = {
            scope:recipient.house ?= {
                any_house_member = { count > 1 }
            }
        }

        is_valid = {
            # Incompatible
            custom_tooltip = {
                text = nfm_int_revoke_banish_exclusive_tt
                scope:eradicate_family = no
            }
        }

        # Tooltip
        current_description = nfm_int_revoke_banish_desc
    }


    on_accept = {
        ## Setup
        # Save their family title
        scope:recipient = {
            random_held_title = {
                limit = { is_noble_family_title = yes }
                save_scope_as = family_title
            }
        }

        # Categorize family into those_to_kill and those_to_banish
        scope:recipient = {
            every_close_or_extended_family_member = {
                limit = {
                    top_liege ?= scope:actor
                    house ?= scope:recipient.house

                    NOT = { is_close_family_of = scope:actor }
                }
                if = {
                    limit = {
                        age >= 16
                        age < 70
                        is_faith_dominant_gender = yes
                        OR = {
                            is_parent_of = scope:recipient
                            is_child_of = scope:recipient
                        }
                    }
                    add_to_list = those_to_kill
                }
                else = {
                    add_to_list = those_to_banish
                }
            }
            # Add spouses to banish list
            every_spouse = {
                limit = {
                    top_liege = scope:actor
                }
                add_to_list = those_to_banish
            }
        }


        ## Calculate Domicile Worth & Distribute Gold accordingly
        ## Sacking Estate grants the Actor more
        nfm_effect_revoke_house_distribute_gold = {
            FAMILY_DOMICILE = scope:family_title.title_domicile
        }

        ## House Relations - Damage relation with the house being destroyed
        nfm_effect_revoke_house_do_house_relations = yes

        ## Destroy the Title
        scope:actor = { destroy_title = scope:family_title }

        ## Family Punishment Options
        # Banish or Eradicate their family
        nfm_effect_revoke_house_punish_family = yes

        ## Post Effects
        # Execute or Banish them
        scope:recipient = {
            if = {
                limit = { scope:eradicate_family = yes }
                death = {
                    death_reason = death_execution
                    killer = scope:actor
                }
            }
            else = {
                limit = { scope:eradicate_family = no }

                # Banish them
                add_opinion = {
                    modifier = banished_me
                    target = scope:actor
                }

                # They might become an adventurer
                hidden_effect = { release_from_prison = yes }
                banish_effect = { BANISHER = scope:actor }

                # If they are an adventurer afterwards.. EXPEL THEM
                if = {
                    limit = { is_landless_ruler = yes }
                    set_variable = {
                        name = adventurer_expeller
                        value = scope:actor
                        months = 4
                    }
                    scope:actor = { save_scope_as = expeller }
                    trigger_event = {
                        id = ep3_interactions_events.0500
                        days = 1
                    }
                    trigger_event = {
                        id = ep3_interactions_events.0501
                        months = 3
                    }
                }
            }
        }

        # Tyranny and Stress Gain
        scope:actor = {
            nfm_effect_revoke_house_stress_tyranny_impacts = yes
        }

    }

    cost = {
        # 200-800, base of 100 and 50 per dynasty level
        prestige = {
            value = 100

            add = {
                value = 50
                multiply = scope:recipient.dynasty.dynasty_prestige_level
            }

            min = 200
            max = 800
        }

        # Match half their influence plus a base cost of 50, 100 if they don't have influence
        influence = {
            value = 50

            if = {
                limit = {
                    scope:recipient = { government_has_flag = government_has_influence }
                }
                add = {
                    value = scope:recipient.influence
                    multiply = 0.5
                }
            }
            else = {
                add = 50
            }

            min = 50
            max = 600
        }
    }
}



############################################
# DISSOLVE DYING NOBLE HOUSE
# Peaceful dissolution for houses that
# have no future (few members, no fertile
# members, all married non-dominantly)
############################################

nfm_int_dissolve_house = {
    icon = dissolve_house
    category = interaction_category_vassal
    common_interaction = yes
    interface_priority = 65
    desc = nfm_int_dissolve_house_desc

    greeting = negative

    is_shown = {
        scope:actor = {
            is_ai = no
            top_liege = this

            # Check Government
            government_allows = noble_families
        }

        scope:recipient = {
            top_liege ?= scope:actor
            is_ruler = yes
            is_landed = no

            # Must hold a noble family title
            any_held_title = {
                is_noble_family_title = yes
            }
        }

        NOT = { scope:actor = scope:recipient }
    }

    is_valid_showing_failures_only = {
        scope:actor = { is_available = yes }
    }

    is_valid = {
        # Must meet dying house criteria
        scope:recipient.house = {
            nfm_trigger_is_dying_house = yes
        }

        # No other house members should be landed in the realm
        scope:recipient.house = {
            custom_description = {
                text = nfm_loc_house_must_be_unlanded
                NOT = {
                    any_house_member = {
                        is_landed = yes
                        top_liege = scope:actor
                        NOT = { this = scope:recipient }
                    }
                }
            }
        }
    }

    auto_accept = {
        custom_description = {
            text = nfm_int_dissolve_house_auto_accept_reason
            always = yes
        }
    }

    on_auto_accept = {
        # Covered by on_accept
    }

    on_accept = {
        ## Setup
        # Save their family title
        scope:recipient = {
            random_held_title = {
                limit = { is_noble_family_title = yes }
                save_scope_as = family_title
            }
        }

        ## Calculate Domicile Worth & Distribute Gold
        # Peaceful dissolution: members keep most of the value
        scope:family_title.title_domicile ?= {
            save_scope_value_as = {
                name = domicile_value
                value = nfm_value_domicileworth
            }
        }

        # Actor gets 10% (administrative fees)
        scope:actor = {
            add_gold = {
                value = scope:domicile_value
                multiply = 0.1
            }
        }

        # Recipient gets 90% (liquidated assets)
        scope:recipient = {
            add_gold = {
                value = scope:domicile_value
                multiply = 0.9
            }
        }

        ## House Relations - Minor damage for peaceful dissolution
        scope:actor.house = {
            change_house_relation_effect = {
                HOUSE = scope:recipient.house
                VALUE = house_relation_damage_minor_value
                REASON = nfm_dissolved_house
                CHAR = scope:actor
                TARGET_CHAR = scope:recipient
                TITLE = scope:family_title
            }
        }

        ## Destroy the Title
        scope:recipient = { destroy_title = scope:family_title }

        ## Post Effects
        # Members stay in the realm - no banishment
        scope:recipient = {
            # Opinion penalty, but less severe than revoke
            add_opinion = {
                modifier = nfm_op_house_dissolved
                target = scope:actor
            }
        }

        # Stress impacts for actor
        scope:actor = {
            # Minimal tyranny - this is a peaceful/merciful action
            add_tyranny = nfm_value_dissolve_house_tyranny_gain
        }
    }

    cost = {
        prestige = {
            value = 25

            add = {
                value = 25
                multiply = scope:recipient.dynasty.dynasty_prestige_level
            }

            min = 50
            max = 400
        }
    }
}
