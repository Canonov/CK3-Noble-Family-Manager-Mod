################################
# Grant Nobility to Lowborn
################################
nfm_int_elevate_lowborn = {
    category = interaction_category_vassal
    common_interaction = no
    icon = icon_dynasty

    desc = nfm_int_elevate_lowborn_desc

    greeting = positive

    interface_priority = 50

    auto_accept = yes

    is_shown = {
        scope:actor = {
            is_ai = no
            is_landed = yes
        }
        scope:recipient = {
            is_lowborn = yes
            is_adult = yes
            is_imprisoned = no
            OR = {
                is_courtier_of = scope:actor
                AND = {
                    exists = employer
                    employer = {
                        OR = {
                            this = scope:actor
                            any_liege_or_above = { this = scope:actor }
                        }
                    }
                }
            }
        }
        NOT = { scope:actor = scope:recipient }
    }

    is_valid_showing_failures_only = {
        scope:recipient = {
            NOT = { has_trait = devoted }
        }
    }

    on_accept = {
        scope:recipient = {
            custom_tooltip = {
                text = nfm_elevate_lowborn_tt
                create_dynasty = { save_scope_as = new_dynasty }
            }
        }

        scope:actor = {
            stress_impact = {
                arrogant = minor_stress_impact_gain
                greedy = minor_stress_impact_gain
                generous = minor_stress_impact_loss
                compassionate = minor_stress_impact_loss
            }

            # Notify the Player
            send_interface_toast = {
                type = event_generic_good_with_text

                title = nfm_toast_elevated_lowborn
                desc = nfm_toast_elevated_lowborn_desc
                tooltip = nfm_toast_elevated_lowborn_tt

                left_icon = scope:recipient
                right_icon = scope:new_dynasty
            }
        }
    }

    cost = {
        prestige = 100
    }
}


################################
# Grant Noble Family Status
################################
nfm_int_grant_house = {
    category = interaction_category_vassal
    common_interaction = no
    icon = icon_shift_privileges

    desc = nfm_int_grant_house_desc

    greeting = positive

    interface_priority = 50

    auto_accept = yes

    is_shown = {
        scope:actor = {
            # for now, maybe
            is_ai = no

            is_landed = yes

            # Must be at least a King
            highest_held_title_tier >= tier_kingdom

            # Check Government
            government_allows = noble_families
        }
        scope:recipient = {
            AND = {
                is_courtier = yes
                employer.top_liege ?= scope:actor
            }
            is_lowborn = no
            is_adult = yes
            is_imprisoned = no

            # Check Marriage Type
            OR = {
                # are unmarried
                is_married = no
                # dominant partner in at least one of their marriages
                AND = {
                    is_female = yes
                    matrilinear_marriage = yes
                }
                AND = {
                    is_male = yes
                    matrilinear_marriage = no
                }
                # Your marriage is same-sex
                NOR = {
                    AND = {
                        any_consort = {
                            is_female = yes
                        }
                        is_male = yes
                    }
                    AND = {
                        any_consort = {
                            is_male = yes
                        }
                        is_female = yes
                    }
                }
            }
        }
        NOT = { scope:actor = scope:recipient }
    }

    is_valid_showing_failures_only = {
        scope:recipient = {
            NOR = {
                has_trait = devoted
                has_trait = gallivanter
            }
            custom_description = {
                text = is_not_theocratic_court_chaplain
                subject = scope:recipient
                NAND = {
                    scope:actor.faith = {
                        has_doctrine = doctrine_theocracy_temporal
                    }
                    scope:recipient = {
                        faith = {
                            has_doctrine = doctrine_theocracy_temporal
                        }
                        has_council_position = councillor_court_chaplain
                    }
                    scope:actor.faith.religious_head != scope:recipient
                }
            }
        }
    }

    on_accept = {
        scope:recipient = {
            custom_tooltip = {
                text = create_noble_family_tt
                create_noble_family_effect = { GOVERNMENT_GIVER = scope:actor }
            }

            ## Vassalize them since creating a new house doesn't anymore apparently
            nfm_effect_vassalize = {
                LIEGE = scope:actor
                VASSAL = scope:recipient
            }

            custom_tooltip = {
                text = nfm_int_grant_house_costs_tt
                add_prestige = 250
                change_influence = 150
                add_gold = 50
            }


            if = {
                limit = {
                    scope:actor = { government_allows = merit }
                }
                add_character_modifier = {
                    modifier = tgp_passed_provincial_exam_modifier
                }
                change_merit_level = 3
            }

            # Opinion boost towards the granter
            add_opinion = {
                modifier = nfm_op_elevated
                target = scope:actor
            }
        }

        # Improve house relation with newly created noble house
        scope:actor.house = {
            change_house_relation_effect = {
                HOUSE = scope:recipient.house
                VALUE = house_relation_improve_medium_value
                REASON = nfm_granted_house
                CHAR = scope:actor
                TARGET_CHAR = scope:recipient
                TITLE = scope:new_title
            }
        }

        scope:actor = {
            stress_impact = {
                paranoid = minor_stress_impact_gain
                greedy = minor_stress_impact_gain
                arrogant = minor_stress_impact_gain
                generous = minor_stress_impact_loss
                trusting = minor_stress_impact_loss
                peasant_leader = major_stress_impact_loss
                populist_leader = major_stress_impact_loss
            }

            # Grant Hook
            if = {
                limit = {
                    scope:recipient.dynasty = dynasty
                }
                # Grant strong loyalty hook to the granter
                add_hook = {
                    target = scope:recipient
                    type = loyalty_hook
                }
            }
            else = {
                # Grant hook to the granter
                add_hook = {
                    target = scope:recipient
                    type = obligation_hook
                }
            }
        }
    }

    cost = {
        prestige = {
            value = 250

            # Vassals pay double
            if = {
                limit = { scope:actor = { is_independent_ruler = no } }
                multiply = 2
            }
        }
        gold = {
            value = 75

            # Treasury Pays 50 Less
            if = {
                limit = {
                    scope:actor = { has_treasury = yes }
                }
                value = 25
            }
        }

        treasury = {
            value = 0

            # Treasury Pays 50
            if = {
                limit = {
                    scope:actor = { has_treasury = yes }
                }
                value = 50
            }
        }

        influence = {
            value = 150

            # Vassals pay double
            if = {
                limit = { scope:actor = { is_independent_ruler = no } }
                multiply = 2
            }
        }

    }
}
